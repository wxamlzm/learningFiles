- 防抖函数，且第一次运行

```
  debounce (func, delay) {
    let timer = null // 定时器并利用闭包保存
    let count = 0
    return function () {
      const context = this // 记录 this 值,防止在回调函数中丢失
      const args = arguments // 函数参数
      // 首次立刻执行
      if (count === 0) {
        count++
        func.apply(context, args)
        timer = setTimeout(() => {
          // 防止 this 值变为 window
          func.apply(context, args)
        }, delay)
      } else {
        count++
        // 如果定时器存在，则清除定时器(如果没有,也没必要进行处理)
        if (timer) {
          clearTimeout(timer)
          timer = null
        }

        timer = setTimeout(() => {
          // 防止 this 值变为 window
          func.apply(context, args)
        }, delay)
      }
    }
  }
```
